{{/* Renders a status badge. Context: dict "status" (string) "statuses" (map of status definitions) */}} {{- $status := .status }} {{- $statuses := .statuses }} {{- with $status }} {{- $def := index $statuses (lower .) }} {{- $color := "" }} {{- $label := . | upper }} {{- with $def }} {{- $color = .color }} {{- $label = .label | upper }} {{- end }} {{ $label }} {{- end }}
