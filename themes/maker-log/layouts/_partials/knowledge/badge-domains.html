{{/* Renders domain badges. Context: dict "domains" (slice of strings) "domainData" (map of domain definitions) */}} {{- $domains := .domains }} {{- $domainData := .domainData }} {{- with $domains }} {{- range . }} {{- $def := index $domainData (lower .) }} {{- $lbl := cond (ne $def nil) (index $def "label") (upper .) }} {{ $lbl }} {{- end }} {{- end }}
