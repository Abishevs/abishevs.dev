{{/* Generic profile-driven Knowledge Node header. Resolves the page's profile and renders title, subtitle, meta, tech badges, and resources. Context: dict "page" (the page object) */}} {{- $page := .page }} {{- $profiles := hugo.Data.knowledge.profiles }} {{- $statuses := hugo.Data.knowledge.statuses }} {{- $domainData := hugo.Data.knowledge.domains }} {{/* ─── Resolve profile ─── */}} {{- $profileKey := "" }} {{- with $page.Params.profile }} {{- $profileKey = . }} {{- else }} {{- $profileKey = index $profiles._defaults $page.Section }} {{- end }} {{- $profile := index $profiles $profileKey }} {{/* ─── Title ─── */}} {{- $titleText := $page.Title }} {{- with $profile.title }} {{- $titleText = index $page.Params . | default $page.Title }} {{- end }}

{{ $titleText }}{{- if $profile.titleSuffix }}{{- $sv := index $page.Params $profile.titleSuffix }}{{- with $sv }} by {{ . }}{{ end }}{{- end }}

{{/* ─── Subtitle fields ─── */}} {{- with $profile.subtitle }}
{{- range . }} {{- $cls := .class }} {{- if eq .field "_duration" }} {{- $page.Params.startDate | time.Format "Jan 2006" }} {{- if $page.Params.current }} – Present {{- else if $page.Params.endDate }} – {{ $page.Params.endDate | time.Format "Jan 2006" }} {{- end }} {{- else }} {{- $val := index $page.Params .field }} {{- with $val }}{{ $val }}{{ end }} {{- end }} {{- end }}
{{- end }} {{/* ─── Meta fields ─── */}} {{- with $profile.meta }}
{{- range . }} {{- $field := .field }} {{- $type := .type }} {{- $label := .label }} {{- $prefix := .prefix }} {{- $domainKey := .domainKey | default "projectType" }} {{- if eq $type "status" }} {{- with index $page.Params $field }} {{ partial "knowledge/badge-status.html" (dict "status" . "statuses" $statuses) }} {{- end }} {{- else if eq $type "domains" }} {{- $raw := index $page.Params $domainKey }} {{- $domains := slice }} {{- with $raw }} {{- if reflect.IsSlice . }}{{- $domains = . }}{{- else }}{{- $domains = slice . }}{{- end }} {{- end }} {{- with $domains }} {{ partial "knowledge/badge-domains.html" (dict "domains" . "domainData" $domainData) }} {{- end }} {{- else if eq $type "badge" }} {{- $val := index $page.Params $field }} {{- with $val }} {{ . | humanize | title }} {{- end }} {{- else if eq $type "flag" }} {{- $val := index $page.Params $field }} {{- if $val }} {{ $label }} {{- end }} {{- else if eq $type "text" }} {{- $val := index $page.Params $field }} {{- with $val }} {{ with $prefix }}{{ . }} {{ end }}{{ $val }} {{- end }} {{- else if eq $type "date" }} {{ with $prefix }}{{ . }} {{ end }}{{ $page.Date.Format "Jan 2006" }} {{- end }} {{- end }}
{{- end }} {{/* ─── Tech badges (linked if technology page exists, sorted: linked first) ─── */}} {{- with $profile.techField }} {{- $techs := index $page.Params . }} {{- with $techs }}
{{- $linked := slice }} {{- $unlinked := slice }} {{- range . }} {{- $techPage := site.GetPage (printf "/technologies/%s" (. | urlize)) }} {{- if $techPage }} {{- $linked = $linked | append (dict "name" . "page" $techPage) }} {{- else }} {{- $unlinked = $unlinked | append . }} {{- end }} {{- end }} {{- range $linked }} {{ .name }} {{- end }} {{- range $unlinked }} {{ . }} {{- end }}
{{- end }} {{- end }} {{/* ─── Links (generic [[links]] in frontmatter) ─── */}} {{- with $page.Params.links }} {{ partial "knowledge/external-links.html" (dict "links" .) }} {{- end }}
